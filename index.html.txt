    <script>
        // Función para calcular la cuenta regresiva
        function countdown() {
            const eventDate = new Date("2025-02-14T00:00:00").getTime();
            const interval = setInterval(() => {
                const now = new Date().getTime();
                const distance = eventDate - now;

                if (distance < 0) {
                    document.getElementById("countdown").innerHTML = "¡Feliz San Valentín! ❤️";
                    clearInterval(interval);
                    return;
                }

                const days = Math.floor(distance / (1000 * 60 * 60 * 24));
                const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((distance % (1000 * 60)) / 1000);

                document.getElementById("countdown").innerHTML = 
                    `${days} días, ${hours} horas, ${minutes} minutos, ${seconds} segundos`;
            }, 1000);
        }

        // Función para abrir el mapa en ventana emergente
        function abrirMapa() {
            let url = "https://maps.google.com/maps/place//data=!4m2!3m1!1s0x8428ae7e76505807:0xa12d6adb4a049726?entry=s&sa=X&ved=1t:8290&hl=es-mx&ictx=111";
            let opciones = "width=800,height=600,scrollbars=yes";
            window.open(url, "_blank", opciones);
        }

        
     // Ejecutar la cuenta regresiva al cargar la página
        window.onload = countdown;
    </script>
